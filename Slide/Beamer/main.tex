\section{Mô hình hóa bài toán}

%====================================================
% SLIDE 1.1: DLA truyền thống và Hướng mở rộng đề xuất
%====================================================
\begin{frame}{1.1 \; DLA truyền thống và Hướng mở rộng đề xuất}
    \begin{block}{DLA truyền thống \cite{zhong2019publaynet, dla_benchmark_2022_sigkdd_2022_doclaynet} — Single-Page, Closed-Set}
        Học hàm $\mathcal{F}(\mathcal{I}) \rightarrow \mathcal{Y} = \{(b_i,\; c_i)\}_{i=1}^{N}$ với:
        \begin{itemize}
            \item $b_i = [x_{\min}, y_{\min}, x_{\max}, y_{\max}]$ — bounding box
            \item $c_i \in \mathcal{C}_{\text{fixed}}$ — tập nhãn \textbf{tĩnh} (Title, Text, Table, Figure,...)
        \end{itemize}
        \textit{Hạn chế:} Phân tích các trang độc lập, thiếu sự liên kết ngữ nghĩa giữa các vùng, tập nhãn đóng.
    \end{block}

    \vspace{0.3em}
    \begin{block}{Graph-based DLA \cite{dla_2022_coling_2022_doc_gcn_heterogeneous_graph_convolutional_networks_for_document, dla_2023_icdar_2023_a_graphical_approach_to_document_layout_analysis_pdf} — Bổ sung quan hệ cấu trúc}
        $\mathcal{Y} = (\mathcal{V},\; \mathcal{E})$ \quad với $\mathcal{V}$ = tập vùng layout, $\mathcal{E}$ = tập quan hệ (reading order, hierarchy).\\
        \textit{Hạn chế:} Vẫn giới hạn trong phạm vi trang đơn (single-page), thiếu ngữ cảnh xuyên trang.
    \end{block}

    \vspace{0.3em}
    \begin{alertblock}{Mô hình bài toán đề xuất — Cross-Page Multi-Page DLA}
        $$\mathcal{F}\!\left(\mathcal{I}_t \;\big|\; \mathcal{I}_{t-k:t-1},\; \mathcal{S}_{t-1}\right) \;\longrightarrow\; \left(\mathcal{Y}_t,\; \mathcal{S}_t\right)$$
        \begin{itemize}
            \item $\mathcal{I}_{t-k:t-1}$: \textbf{Visual context} từ $k$ trang liền trước.
            \item $\mathcal{S}_{t-1}$: \textbf{Trailing state} — trạng thái cấu trúc tích lũy từ \textit{tất cả} trang trước đó ($1 \ldots t{-}1$).
            \item $\mathcal{Y}_t = (\mathcal{V}_t, \mathcal{E}_t)$: Layout graph chứa \textbf{inter-page relations} và tính liên tục (continuation).
        \end{itemize}
    \end{alertblock}
\end{frame}

%====================================================
% SLIDE 1.2: Thông tin layout nội bộ trang
%====================================================
\begin{frame}{1.2 \; Thông tin layout nội bộ một trang (Intra-Page)}
    \begin{block}{DLA truyền thống — Phân tích từng trang độc lập}
        Với mỗi ảnh trang $\mathcal{I}$, trích xuất tập thực thể layout:
        $\quad \mathcal{F}(\mathcal{I}) \rightarrow \mathcal{Y} = \{(b_i,\; c_i)\}_{i=1}^{N}$
    \end{block}

    \vspace{0.3em}
    Mỗi phần tử layout trên một trang bao gồm các thông tin sau:

    \vspace{0.2em}
    \begin{table}[]
        \centering
        \resizebox{\textwidth}{!}{
        \begin{tabular}{lp{5cm}p{6cm}}
        \toprule
        \textbf{Thông tin} & \textbf{Ý nghĩa} & \textbf{Ví dụ} \\
        \midrule
        $b_i$ = Bounding box & Vị trí vật lý: $[x_{\min}, y_{\min}, x_{\max}, y_{\max}]$ & Hình chữ nhật bao quanh một đoạn văn, bảng, hình \\
        \midrule
        $c_i$ = Category & Loại thành phần layout & Title, Section\_Heading, Text\_Block, Table, Figure, List, Header, Footer,... \\
        \midrule
        Content snippet & Vài chục ký tự đầu tiên & ``3.1 TERMS AND DEFINITIONS...'' \\
        \midrule
        \multicolumn{3}{l}{\textit{Các thông tin trên đủ cho single-page. Nhưng khi mở rộng sang multi-page, cần thêm:}} \\
        \midrule
        $\text{parent\_id}$ & Quan hệ cha-con (hierarchy) & Paragraph 1.1.1 $\rightarrow$ thuộc Section\_Heading 1.1 \\
        \midrule
        $\text{node\_id}$ & Định danh toàn cục duy nhất & \texttt{node\_045} — không trùng qua mọi trang \\
        \midrule
        $\text{continuation}$ & Liên kết xuyên trang & ``Nối tiếp \texttt{node\_012} từ trang trước'' \\
        \bottomrule
        \end{tabular}
        }
    \end{table}

    \vspace{0.2em}
    {\small \textit{3 dòng cuối (parent\_id, node\_id, continuation) là phần} \textbf{mở rộng của chúng ta} \textit{so với DLA truyền thống.}}
\end{frame}

%====================================================
% SLIDE 1.3: Phân tích đa trang là gì?
%====================================================
\begin{frame}{1.3 \; Phân tích đa trang là gì? (Cross-Page DLA)}
    \begin{alertblock}{Bài toán mới: Để phân tích trang $t$, cần ngữ cảnh từ các trang trước}
        $$\mathcal{F}\!\left(\mathcal{I}_t \;\big|\; \underbrace{\mathcal{I}_{t-k:t-1}}_{\text{ảnh $k$ trang trước}},\; \underbrace{\mathcal{S}_{t-1}}_{\text{bộ nhớ tích lũy}}\right) \;\longrightarrow\; \left(\underbrace{\mathcal{Y}_t}_{\text{layout trang $t$}},\; \underbrace{\mathcal{S}_t}_{\text{bộ nhớ cập nhật}}\right)$$
    \end{alertblock}

    \vspace{0.2em}
    \textbf{Tại sao phải phân tích đa trang?} Ba tình huống mà single-page bất lực:

    \vspace{0.2em}
    \begin{columns}[T]
        \column{0.32\textwidth}
        \begin{exampleblock}{\small (a) Category phụ thuộc ngữ cảnh}
            {\scriptsize
            Trang 8 chứa danh sách ``3.1.10 CALIBRATION, 3.1.11 CAPABILITY,...''\\[0.3em]
            Nếu \textbf{chỉ nhìn trang 8}: đây là ``List\_Block'' hay ``Definition\_List''?\\[0.3em]
            \textbf{Cần biết:} Trang 7 có heading ``Section 3: Terms and Definitions'' $\rightarrow$ đây là Definition\_List.
            }
        \end{exampleblock}

        \column{0.32\textwidth}
        \begin{exampleblock}{\small (b) Nội dung bị cắt ngang trang}
            {\scriptsize
            Trang 3: Mục lục (ToC) hiển thị đến mục 8.5...\\[0.3em]
            Trang 4: Tiếp tục từ mục 8.5 đến phụ lục.\\[0.3em]
            \textbf{Cần biết:} Đây là \textit{một} ToC bị ngắt, không phải hai bảng riêng biệt.
            }
        \end{exampleblock}

        \column{0.32\textwidth}
        \begin{exampleblock}{\small (c) Hierarchy xuyên trang}
            {\scriptsize
            Section 1 ``Scope'' bắt đầu ở trang 5.\\
            Paragraphs 1.1--1.7 kéo dài đến trang 6.\\[0.3em]
            \textbf{Cần biết:} Paragraphs ở trang 6 thuộc Section 1 (ở trang 5), không phải top-level.
            }
        \end{exampleblock}
    \end{columns}

    \vspace{0.4em}
    \begin{center}
        $\Rightarrow$ \textbf{Output} không chỉ là tập bbox + category, mà là \textbf{đồ thị cấu trúc} $\mathcal{Y}_t = (\mathcal{V}_t, \mathcal{E}_t)$ với các liên kết \textit{xuyên trang}.
    \end{center}
\end{frame}

%====================================================
% SLIDE 1.4: Bộ nhớ (Trailing State) mô hình hóa thế nào
%====================================================
\begin{frame}{1.4 \; Bộ nhớ tích lũy $\mathcal{S}_t$ — Mô hình hóa chi tiết}
    \textbf{Bộ nhớ} $\mathcal{S}_t$ là một cấu trúc dữ liệu được \textbf{cập nhật sau mỗi trang} và \textbf{truyền sang trang kế tiếp}.\\
    Tương tự \textit{hidden state} $h_t$ trong mạng hồi quy (RNN): $\;h_t = f(x_t, h_{t-1})$.

    \vspace{0.4em}
    \begin{table}[]
        \centering
        \resizebox{\textwidth}{!}{
        \begin{tabular}{lp{4.5cm}p{7cm}}
        \toprule
        \textbf{Trường trong $\mathcal{S}_t$} & \textbf{Lưu trữ gì} & \textbf{Ví dụ (sau khi phân tích trang 7)} \\
        \midrule
        \texttt{last\_node\_id} & Bộ đếm ID toàn cục & \texttt{``node\_089''} $\rightarrow$ trang 8 bắt đầu từ \texttt{node\_090} \\
        \midrule
        \texttt{active\_parent\_stack} & Ngăn xếp các heading đang ``mở'' (ngoài $\rightarrow$ trong) &
            \texttt{[} \\
            & & \texttt{\ \ \{id: node\_060, cat: Section\_Heading, text: ``3. Terms''\},} \\
            & & \texttt{\ \ \{id: node\_061, cat: Section\_Header, text: ``3.1 Definitions''\}} \\
            & & \texttt{]} \\
            & & $\rightarrow$ Trang 8: mọi definition entry có \texttt{parent = node\_061} \\
        \midrule
        \texttt{element\_cut\_off} & Phần tử bị cắt cuối trang &
            \texttt{\{id: node\_089, cat: Text\_Block, text: ``...chưa hết câu''\}} \\
            & & $\rightarrow$ Trang 8: phần tử đầu tiên sẽ mang \texttt{continues = node\_089} \\
        \midrule
        \texttt{new\_categories} & Nhãn mới phát sinh & \texttt{[``Definition\_Entry'': lý do...]} — bổ sung vào vocabulary \\
        \bottomrule
        \end{tabular}
        }
    \end{table}

    \vspace{0.3em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        {\small \textbf{Visual context} $\mathcal{I}_{t-k:t-1}$ (ảnh trang kề):
        \begin{itemize}
            \item Cho biết trang trông \textbf{như thế nào}
            \item Giới hạn: chỉ ``nhìn'' được $k$ trang gần nhất
        \end{itemize}}

        \column{0.48\textwidth}
        {\small \textbf{Bộ nhớ} $\mathcal{S}_{t-1}$ (trailing state):
        \begin{itemize}
            \item Cho biết trang đó \textbf{nằm ở đâu} trong cấu trúc tài liệu
            \item Tích lũy từ \textit{tất cả} trang trước — không bị giới hạn bởi $k$
        \end{itemize}}
    \end{columns}
\end{frame}

%====================================================
% SLIDE: Khó khăn — Model
%====================================================
\begin{frame}{2.2 \; Hạn chế của các phương pháp SOTA hiện hành}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{\color{dangerred} Vision-based (YOLO \cite{dla_2024_arxiv_doclayout_yolo_enhancing_document_layout_analysis_through_diverse_syn}, RT-DETR \cite{zhao2023detrs})}
            \textbf{Ưu điểm:} Bbox chính xác, tốc độ cao.\\
            \textbf{Giới hạn:}
            \begin{itemize}
                \item Hiệu suất suy giảm mạnh đối với tập nhãn ngoài huấn luyện (Out-of-Distribution).
                \item Không thể xác định nhãn nếu thiếu ngữ cảnh từ các trang trước.
                \item Thiếu khả năng suy luận logic (reasoning) đa trang.
            \end{itemize}
        \end{block}

        \column{0.48\textwidth}
        \begin{block}{\color{dangerred} MLLM (Gemini \cite{gemini_team_2024_gemini}, Qwen3-VL \cite{qwen_team_2025_qwen3vl_thinking})}
            \textbf{Ưu điểm:} Năng lực reasoning mạnh, open-vocabulary.\\
            \textbf{Giới hạn:}
            \begin{itemize}
                \item Suy giảm độ chính xác tọa độ (Grounding degradation) khi trang có mật độ thông tin dày đặc.
                \item Chi phí xử lý token cho ngữ cảnh đa trang quá lớn.
                \item Suy luận End-to-End thường dẫn đến hiện tượng phân mảnh quá mức (over-fragmentation) hoặc bỏ sót (under-detection).
            \end{itemize}
        \end{block}
    \end{columns}

    \vspace{0.4em}
    \begin{block}{\color{dangerred} Parser Pipelines (PaddleOCR \cite{du2020ppocr}, MinerU \cite{wang2024mineru})}
        Hoạt động ổn định trên tài liệu cơ bản, nhưng \textbf{kém hiệu quả} trên layout phức tạp: không ghép nối được các khối văn bản xuyên trang, thiếu cấu trúc phân cấp (hierarchy) rõ ràng và không biểu diễn được quan hệ inter-page.
    \end{block}
\end{frame}

%====================================================
% SLIDE: Khó khăn — Data
%====================================================
\begin{frame}{2.3 \; Thách thức về sự thiếu hụt dữ liệu (Data Scarcity)}
    \begin{alertblock}{Vấn đề cốt lõi}
        Do đây là bài toán mới, cộng đồng hiện chưa có:
        \begin{itemize}
            \item Benchmark đánh giá DLA đa trang (cross-page) có gán nhãn phân cấp (hierarchical annotation).
            \item Bộ dữ liệu huấn luyện (Training dataset) đủ lớn cho bài toán này.
            \item Chi phí tạo Ground truth thủ công là cực kỳ đắt đỏ.
        \end{itemize}
    \end{alertblock}

    \vspace{0.4em}
    \textbf{Đánh đổi (Trade-off) khi sử dụng MLLM làm Data Engine:}
    \begin{itemize}
        \item MLLM dự đoán Bounding box \textbf{kém chính xác hơn} mô hình thuần Vision (như DocLayout-YOLO \cite{dla_2024_arxiv_doclayout_yolo_enhancing_document_layout_analysis_through_diverse_syn}).
        \item Tuy nhiên, mô hình Vision \textbf{không thể} sinh ra logic phân cấp (hierarchy), quan hệ xuyên trang.
        \item $\Rightarrow$ \textbf{Chấp nhận đánh đổi độ chính xác tuyệt đối của bbox}, tập trung khai thác giá trị \textit{chú giải cấu trúc} (structural annotation: hierarchy, relation, cross-page).
        \item Việc tinh chỉnh (refinement) tọa độ bbox sẽ được tối ưu bằng các mô hình hậu xử lý (post-processing) ở giai đoạn sau.
    \end{itemize}

    \vspace{0.3em}
    \textbf{$\Rightarrow$ Giải pháp:} Xây dựng \textbf{Data Engine Pipeline} hiệu quả dựa trên MLLM SOTA nhằm tự động hóa việc tạo Benchmark và tập dữ liệu huấn luyện.
\end{frame}

\begin{frame}{4.2 \; Pha 1 — Khám phá Nhãn ngữ nghĩa (Category Discovery)}
    \begin{columns}[T]
        \column{0.55\textwidth}
        \textbf{Cơ chế hoạt động:}
        \begin{itemize}
            \item Phân tích tài liệu theo cửa sổ trượt (sliding window) kích thước $k$ trang.
            \item MLLM \textbf{chỉ} suy luận về Category, \textbf{không} dự đoán tọa độ (localization).
            \item Đầu ra: Danh sách Category kèm định nghĩa chi tiết \& mục đích sử dụng (downstream purpose).
            \item Hợp nhất (Merge) \& loại bỏ trùng lặp (Deduplicate) từ các cửa sổ.
        \end{itemize}

        \vspace{0.5em}
        \textbf{Vai trò của thiết kế này:}
        \begin{itemize}
            \item Đảm bảo nhãn mang tính \textbf{thực tiễn} (Macro-level).
            \item Duy trì tính \textbf{nhất quán} ngữ nghĩa qua toàn bộ tài liệu.
            \item Thu hẹp không gian tìm kiếm (search space) cho Pha 2, hạn chế hiện tượng ảo giác (hallucination) tạo ra nhãn ảo.
        \end{itemize}

        \column{0.42\textwidth}
        \begin{exampleblock}{Kết quả minh họa (Quality Manual)}
            {\scriptsize
            13 Category được trích xuất:
            \begin{itemize}
                \item \texttt{Title}, \texttt{Section\_Heading}
                \item \texttt{Section\_Header}, \texttt{Text\_Block}
                \item \texttt{Table\_of\_Contents}
                \item \texttt{Table}, \texttt{List\_Block}
                \item \texttt{Diagram}, \texttt{Image}, \texttt{Logo}
                \item \texttt{Metadata\_Block}
                \item \texttt{Header}, \texttt{Footer}
            \end{itemize}
            Mỗi category được đính kèm \texttt{downstream\_purpose} làm ngữ cảnh cho Pha 2.
            }
        \end{exampleblock}
    \end{columns}
\end{frame}

\begin{frame}{4.3 \; Pha 2 — Phân tích Cấu trúc (Structural Parsing)}
    \begin{columns}[T]
        \column{0.52\textwidth}
        \textbf{Pha 2.1 — Khởi tạo (Trang 1):}
        \begin{itemize}
            \item Phân tích dựa trên Visual input và tập Categories từ Pha 1.
            \item Khởi tạo Trailing state $S_0$:
            \begin{itemize}
                \item \texttt{last\_assigned\_node\_id}: \texttt{node\_000}
                \item \texttt{active\_parent\_stack}: \texttt{[]}
                \item \texttt{element\_cut\_off...}: \texttt{null}
            \end{itemize}
        \end{itemize}

        \vspace{0.4em}
        \textbf{Pha 2.2 — Cập nhật tăng dần (Trang $2..n$):}
        \begin{itemize}
            \item Đầu vào: \textit{Ảnh trang $t-1$} + \textit{Ảnh trang $t$} + \textit{Trailing state $S_{t-1}$} + \textit{Categories}.
            \item Đầu ra: Bbox, node\_id, category, parent\_node\_id, continuation\_flag.
            \item Cập nhật $S_{t}$ để truyền tiếp cho trang $t+1$.
        \end{itemize}

        \column{0.45\textwidth}
        \textbf{Xử lý liên kết xuyên trang (Cross-page):}
        \begin{exampleblock}{\small Ví dụ 1: Ngắt trang bảng mục lục}
            {\scriptsize
            Trang 3: \texttt{node\_012} (Table\_of\_Contents)\\
            $\rightarrow$ Ghi nhận cắt ngang: \texttt{node\_012}\\[0.3em]
            Trang 4: \texttt{node\_013} (Table\_of\_Contents)\\
            $\rightarrow$ Nhận diện tiếp nối: \texttt{node\_012}\\[0.3em]
            \textit{Pha 3 sẽ thực hiện hợp nhất hai node này.}
            }
        \end{exampleblock}

        \vspace{0.3em}
        \begin{exampleblock}{\small Ví dụ 2: Cấu trúc phân cấp đa trang}
            {\scriptsize
            \texttt{active\_parent\_stack}:\\
            \texttt{[Section 3: ``Terms'',}\\
            \texttt{\ node\_025: ``3.1 Quality...'']  }\\[0.3em]
            $\rightarrow$ Hệ thống duy trì thông tin phân cấp (hierarchy) liên tục, dù đoạn văn nằm ở cách xa nhiều trang.
            }
        \end{exampleblock}
    \end{columns}
\end{frame}

\begin{frame}{4.4 \; Pha 3 — Cấu trúc hóa Đồ thị (Tree Assembly)}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textbf{Quy trình (Rule-based):}
        \begin{enumerate}
            \item \textbf{Ghép nối (Merge continuation):}\\ Xử lý chuỗi \texttt{continues\_previous\_node}, nối snippet, cập nhật tham chiếu.
            \item \textbf{Xây dựng cây (Build Tree):}\\ Khôi phục đồ trúc dựa trên \texttt{parent\_node\_id}.
            \item \textbf{Xuất dữ liệu (Export):}\\ Định dạng JSON tree, Text outline, và Mermaid diagram.
        \end{enumerate}

        \vspace{0.3em}
        \textbf{Lý do sử dụng Rule-based thay vì MLLM:}
        \begin{itemize}
            \item Dữ liệu từ Pha 2 đã đảm bảo tính toàn vẹn ngữ nghĩa.
            \item Đảm bảo tính \textbf{tất định (deterministic)}, tối ưu chi phí tính toán, độ trễ thấp (low latency).
        \end{itemize}

        \column{0.48\textwidth}
        \begin{exampleblock}{Kết quả — Quality Manual (11 trang)}
            {\scriptsize
            \begin{itemize}
                \item Xây dựng thành công 136 nodes, 13 categories, độ sâu phân cấp tối đa 3.
                \item Giải quyết chính xác ngắt trang Mục lục (trang 3$\rightarrow$4).
                \item Section 3 ``Terms \& Definitions'': Quản lý 64 định nghĩa con trải dài qua 5 trang dưới cùng 1 node cha.
            \end{itemize}
            }
        \end{exampleblock}
        \vspace{0.2em}
        \begin{exampleblock}{Kết quả — ISO Generic Manual (7 trang)}
            {\scriptsize
            \begin{itemize}
                \item Xây dựng 81 nodes, 12 categories.
                \item Giải quyết trọn vẹn List continuation xuyên trang.
                \item Phân tích chính xác bảng lồng ghép (nested tables).
            \end{itemize}
            }
        \end{exampleblock}
    \end{columns}
\end{frame}

\begin{frame}{4.5 \; Mã giả — Kiến trúc tổng thể Pipeline}
    {\small
    \begin{algorithmic}
    \STATE \textbf{Input:} $D = \{I_1, I_2, \ldots, I_n\}$ \hfill // Tập $n$ ảnh trang tài liệu
    \STATE \textbf{Output:} $T$ = Document Tree \hfill // Đồ thị cấu trúc tích hợp cross-page links
    \STATE
    \STATE \textcolor{warnorange}{\textbf{--- Pha 1: Category Discovery ---}}
    \STATE $\mathcal{C} \leftarrow \emptyset$
    \FOR{mỗi cửa sổ trượt $w$ gồm $k$ trang}
        \STATE $\mathcal{C}_w \leftarrow \text{MLLM}(w, \text{CATEGORY\_PROMPT})$
        \STATE $\mathcal{C} \leftarrow \text{MERGE}(\mathcal{C}, \mathcal{C}_w)$
    \ENDFOR
    \STATE
    \STATE \textcolor{successgreen}{\textbf{--- Pha 2.1: Initial Analysis ---}}
    \STATE $S_0 \leftarrow \text{INIT\_STATE}()$ \hfill // Thiết lập trạng thái khởi tạo
    \STATE $(Y_1, S_1) \leftarrow \text{MLLM}(I_1, \text{null}, S_0, \mathcal{C}, \text{PARSE\_PROMPT})$
    \STATE
    \STATE \textcolor{successgreen}{\textbf{--- Pha 2.2: Incremental Loop ---}}
    \FOR{$t = 2$ \textbf{to} $n$}
        \STATE $(Y_t, S_t) \leftarrow \text{MLLM}(I_t, I_{t-1}, S_{t-1}, \mathcal{C}, \text{PARSE\_PROMPT})$
        \hfill // Cập nhật và truyền Trailing State
    \ENDFOR
    \STATE
    \STATE \textcolor{purple}{\textbf{--- Pha 3: Tree Assembly (Rule-based) ---}}
    \STATE Merge continuation nodes $\rightarrow$ Build tree từ \texttt{parent\_node\_id} $\rightarrow$ Export Format
    \end{algorithmic}
    }
\end{frame}

        \node[box, fill=red!10, draw=dangerred, thick, left=1.0cm of finetune] (distill) {\textbf{Knowledge Distillation}\\ Nén tri thức mô hình};
        \node[box, fill=gray!10, draw=gray, thick, left=1.0cm of distill] (prod) {\textbf{Deployment}\\ Mô hình tối ưu hóa cho môi trường Local};

        \draw[arrow] (collect) -- (engine);
        \draw[arrow] (engine) -- (data);
        \draw[arrow] (data) -- (finetune);
        \draw[arrow] (finetune) -- (distill);
        \draw[arrow] (distill) -- (prod);
    \end{tikzpicture}
    }
    \end{center}

    \vspace{0.3em}
    \textbf{Định hướng triển khai:}
    \begin{enumerate}
        \item Thu thập tập tài liệu PDF đặc thù mà các công cụ hiện hành chưa xử lý được.
        \item Khai thác năng lực MLLM SOTA (Gemini 3.1 Pro \cite{gemini_team_2024_gemini}) làm Data Engine để tổng hợp nhãn (benchmark \& training data).
        \item Fine-tune các mô hình VLM/MLLM nhỏ hơn dựa trên tập dữ liệu tổng hợp này.
        \item Thực hiện Knowledge Distillation nhằm tối ưu hóa chi phí tính toán cho ứng dụng thực tế.
    \end{enumerate}
\end{frame}

\begin{frame}{5.2 \; Phân tích các Chiến lược Fine-tuning}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \begin{block}{Chiến lược 1: Pipeline đa mô hình (Multi-VLM)}
            Huấn luyện nhiều mô hình nhỏ, chuyên biệt hóa cho từng pha:
            \begin{itemize}
                \item VLM\textsubscript{1}: Đảm nhận Category reasoning.
                \item VLM\textsubscript{2}: Trích xuất cấu trúc \& Localization (Bbox).
                \item Module Rule-based xử lý Tree assembly.
            \end{itemize}
            \textit{Ưu điểm:} Tính module hóa cao, dễ cô lập lỗi (debug) và thay thế thành phần.\\
            \textit{Hạn chế:} Độ phức tạp khi triển khai toàn hệ thống cao.
        \end{block}

        \column{0.48\textwidth}
        \begin{block}{Chiến lược 2: Mô hình End-to-End (Single-VLM)}
            Huấn luyện một VLM duy nhất đảm nhận toàn bộ quy trình:
            \begin{itemize}
                \item Đầu vào: Chuỗi ảnh PDF + Instruction.
                \item Đầu ra trực tiếp: Category, Bbox, Hierarchy và Inter-page relations.
            \end{itemize}
            \textit{Ưu điểm:} Kiến trúc gọn gàng, tối ưu hóa quá trình suy luận (inference).\\
            \textit{Hạn chế:} Đòi hỏi mô hình có dung lượng tham số đủ lớn, khó kiểm soát hành vi mô hình.
        \end{block}
    \end{columns}

    \vspace{0.5em}
    \begin{center}
        \textbf{Mục tiêu dài hạn:} Khai thác Knowledge Distillation từ MLLM SOTA để xây dựng mô hình VLM cục bộ (local deployment), loại bỏ sự phụ thuộc vào các API thương mại.
    \end{center}
\end{frame}
