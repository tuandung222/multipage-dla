\section{Giới thiệu (Introduction)}

% --- Slide 1.1: Định nghĩa & Mô hình hóa toán học (truyền thống) ---
\begin{frame}{1.1 \; Định nghĩa \& Mô hình hóa Toán học}
    \begin{block}{Định nghĩa}
        Document Layout Analysis (DLA) là bài toán phát hiện và phân loại tự động các thành phần cấu trúc trong một trang tài liệu: \textit{tiêu đề (Title), đoạn văn (Text), bảng biểu (Table), hình ảnh (Figure), công thức (Formula),...}
    \end{block}

    \vspace{0.3em}
    \textbf{Formulate truyền thống — Multiple-Object Detection / Segmentation:}

    Học hàm $\mathcal{F}(\mathcal{I}) \rightarrow \mathcal{Y}$ ánh xạ từ ảnh tài liệu $\mathcal{I}$ sang tập các thực thể bố cục:
    $$\mathcal{Y} = \bigl\{ (b_i,\; c_i,\; m_i) \bigr\}_{i=1}^{N}$$
    \begin{itemize}
        \item $b_i = [x_{\min}, y_{\min}, x_{\max}, y_{\max}]$ — Tọa độ \textbf{Bounding Box}.
        \item $c_i \in \{\textit{Title, Text, Table, Figure, Formula...}\}$ — \textbf{Nhãn phân loại}.
        \item $m_i \in \{0,1\}^{H \times W}$ — \textbf{Mặt nạ phân đoạn} (Instance Segmentation) cho bố cục phức tạp.
    \end{itemize}

    \vspace{0.3em}
    \begin{columns}[T]
        \column{0.48\textwidth}
        \textit{Ưu điểm:} Đơn giản, tốc độ nhanh, dễ tích hợp.
        \column{0.48\textwidth}
        \textit{Hạn chế:} Không nắm bắt được quan hệ giữa các vùng (thứ tự đọc, phân cấp cấu trúc).
    \end{columns}
\end{frame}

% --- Slide 1.2: Mở rộng Formulation hiện đại ---
\begin{frame}{1.2 \; Mở rộng Formulation — Xu hướng 2024-2025}
    \textbf{Hai hướng mở rộng bổ sung hiện đang được cộng đồng nghiên cứu tích cực:}

    \vspace{0.5em}
    \begin{block}{(A) Graph-based Formulation — Phân tích Cấu trúc \& Quan hệ}
        Không chỉ tìm thực thể (Nodes), mà phải học mối quan hệ (Edges) giữa chúng:
        $$\mathcal{Y} = (\mathcal{V},\; \mathcal{E})$$
        Trong đó: $\mathcal{V}$ = tập các vùng layout; $\mathcal{E}$ = ma trận quan hệ $r_{ij}$ biểu diễn
        \textit{thứ tự đọc (Reading Order)}, \textit{phân cấp cấu trúc (Hierarchical Tree)}.\\
        \textit{Mô hình tiêu biểu: Doc-GCN \cite{dla_2022_coling_2022_doc_gcn_heterogeneous_graph_convolutional_networks_for_document}, VGT (Vision Grid Transformer) \cite{dla_2023_iccv_2023_vision_grid_transformer_for_document_layout_analysis_pdf}.}
    \end{block}

    \vspace{0.4em}
    \begin{block}{(B) Seq2Seq Formulation — LLM / MLLM Paradigm (OCR-Free)}
        Coi DLA như bài toán \textbf{sinh ngôn ngữ có điều kiện} từ ảnh:
        $$P(Y \mid \mathcal{I}) = \prod_{t=1}^{T} P\!\left(y_t \mid y_{<t},\; \mathcal{I}\right)$$
        Chuỗi đầu ra $Y$ chứa cả nội dung \textit{và} tọa độ, ví dụ:\\
        {\small\texttt{[Title][loc\_10\_20\_100\_50] Báo cáo tài chính [End\_Title] ...}}\\
        \textit{Mô hình tiêu biểu: Donut \cite{kim2022donut}, Nougat \cite{nougat_2023}, mPLUG-DocOwl \cite{ye2023mplugdocowl}, Qwen-VL \cite{qwen2vl_2023}.}
    \end{block}
\end{frame}

\section{Literature Review}
\begin{frame}{2.1 \; Các phương pháp tiếp cận hiện nay}
    \begin{itemize}
        \item \textbf{Phương pháp dựa trên thị giác máy tính (CV-based):}
        \begin{itemize}
            \item Coi tài liệu như một hình ảnh thông thường.
            \item Sử dụng các mô hình Object Detection như Mask R-CNN \cite{he2017mask}, YOLO, DETR \cite{carion2020end} \cite{dla_2023_arxiv_document_ai_a_comparative_study_of_transformer_based_graph_based_mode}.
            \item \textit{Ưu điểm}: Tốc độ nhận diện nhanh. \textit{Hạn chế}: Dễ bị nhầm lẫn nếu không có thông tin ngữ nghĩa (text).
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{Phương pháp đa phương thức (Multimodal):}
        \begin{itemize}
            \item Kết hợp thông tin từ Hình ảnh (Image), Văn bản (OCR) và Tọa độ không gian (Bounding box/Layout).
            \item Các mô hình tiêu biểu: Dòng họ LayoutLM (v1, v2, v3) \cite{huang2022layoutlmv3}, DocTr \cite{liao2023doctr}.
        \end{itemize}
        \vspace{0.2cm}
        \item \textbf{Vision-Language Models (VLM) / Generative AI:}
        \begin{itemize}
            \item Tiếp cận End-to-end: Đưa ảnh tài liệu vào và sử dụng LLM để sinh ra trực tiếp mã đánh dấu (Markdown/HTML).
            \item Các mô hình tiên phong: Nougat \cite{nougat_2023}, Donut \cite{kim2022donut}, Qwen-VL \cite{qwen2vl_2023}, GPT-4o.
        \end{itemize}
    \end{itemize}
\end{frame}

\begin{frame}{2.2 \; Xu hướng mới: Kết hợp MLLM và DLA}
    Sự dịch chuyển từ việc chỉ phát hiện vật lý các vùng văn bản (Physical Layout) sang việc hiểu biết cấu trúc ngữ nghĩa phức tạp dưới sự hỗ trợ của LLM/MLLM được chia thành 4 trường phái chính:

    \vspace{0.2cm}
    \begin{table}[]
        \centering
        \resizebox{\textwidth}{!}{
        \begin{tabular}{lp{5cm}p{5.5cm}}
        \toprule
        \textbf{Trường phái} & \textbf{Cách tiếp cận} & \textbf{Mô hình tiêu biểu} \\
        \midrule
        \textbf{1. Layout-as-Tokens} & Ánh xạ Bounding Box thành Token/Embedding, ghép xen kẽ với Text trong chuỗi Prompt để đưa vào LLM. & \textbf{LayTextLLM / LayTokenLLM} \cite{lu2024bounding} (chia sẻ chung Position ID để tiết kiệm ngữ cảnh). \\
        \midrule
        \textbf{2. Layout-as-Modality} & Xem tọa độ như một phương thức độc lập. Sửa đổi trực tiếp kiến trúc Attention hoặc dùng Instruction Tuning. & \textbf{DocLLM} \cite{wang2024docllm} (Disentangled Spatial Attention), \textbf{LayoutLLM} \cite{luo2024layoutllm} (Layout Chain-of-Thought). \\
        \midrule
        \textbf{3. Decoupled \& RAG} & Tách rời DLA chuyên dụng để detect Bounding Box làm ``điểm neo'' (anchors) trước khi đưa vào LLM/RAG. & \textbf{MinerU 2.5} \cite{wang2024mineru} (DocLayout-YOLO \cite{dla_2024_arxiv_doclayout_yolo_enhancing_document_layout_analysis_through_diverse_syn} làm tiền xử lý để tiết kiệm chi phí cho Attention của LLM). \\
        \midrule
        \textbf{4. MLLM End-to-End} & Nhận trực tiếp ảnh (OCR-Free), dùng sức mạnh của LLM tự sinh ra cấu trúc định dạng (Markdown/JSON) & \textbf{mPLUG-DocOwl} \cite{ye2023mplugdocowl}, \textbf{Donut} \cite{kim2022donut}, \textbf{Nougat} \cite{nougat_2023} \\
        \bottomrule
        \end{tabular}
        }
    \end{table}
\end{frame}

\begin{frame}{2.3 \; Quy trình làm Data cho DLA (Training-ready)}
    \begin{block}{Pipeline đề xuất cho dữ liệu huấn luyện}
                	\textbf{B1. Thu thập theo domain} (paper, invoice, legal, manual) →
                	\textbf{B2. Chuẩn hóa annotation} (label schema + format COCO/JSON) →
                	\textbf{B3. Chia train/val/test} theo domain và độ khó →
                	\textbf{B4. Quality control} (sửa lỗi box, class noise, overlap) →
                	\textbf{B5. Huấn luyện baseline + error analysis}.
    \end{block}

        \vspace{0.4em}
        	\textbf{Nguồn dữ liệu theo 3 mức độ giám sát:}
    \begin{itemize}
        \item \textbf{Auto-annotated (quy mô lớn):} PubLayNet, DocBank\cite{zhong2019publaynet,li2020docbank}.
        \item \textbf{Human-annotated (độ chính xác cao):} DocLayNet\cite{dla_benchmark_2022_sigkdd_2022_doclaynet_a_large_human_annotated_dataset_for_documen}.
        \item \textbf{Synthetic augmentation:} DocSynth-300K trong DocLayout-YOLO để tăng đa dạng bố cục\cite{dla_2024_arxiv_doclayout_yolo_enhancing_document_layout_analysis_through_diverse_syn}.
    \end{itemize}

        \vspace{0.3em}
        	\textit{Thông điệp chính: dữ liệu tốt (schema rõ + split chuẩn + QC) thường quan trọng không kém kiến trúc mô hình.}
\end{frame}

\begin{frame}{2.5 \; Benchmark mới (2024--2025)}
    \small
    \begin{itemize}
        \item \textbf{Robustness benchmark (RoDLA):} đánh giá độ bền mô hình dưới nhiều nhiễu/perturbation thay vì chỉ ảnh ``sạch''\cite{dla_2024_cvpr_2024_rodla_benchmarking_the_robustness_of_document_layout_analysis_mod}.
        \item \textbf{Structural error benchmark (LED):} chẩn đoán lỗi cấu trúc đặc thù của layout (merge/split/hallucinated regions)\cite{dla_2025_arxiv_led_benchmark_diagnosing_structural_layout_errors_for_document_layout}.
        \item \textbf{Source-free domain adaptation (SFDLA):} đánh giá khả năng adapt sang miền mới khi không truy cập dữ liệu nguồn ban đầu\cite{dla_2025_arxiv_sfdla_source_free_document_layout_analysis_pdf}.
        \item \textbf{Data-centric scaling:} tăng chất lượng data qua synthetic data (DocSynth) và pipeline unified data construction\cite{dla_2024_arxiv_doclayout_yolo_enhancing_document_layout_analysis_through_diverse_syn,dla_2025_arxiv_pp_doclayout_a_unified_document_layout_detection_model_to_accelerate_}.
    \end{itemize}

    \vspace{0.4em}
    \begin{block}{Kết luận review}
        Xu hướng benchmark mới dịch chuyển từ \textit{``mAP cao trên test chuẩn''} sang \textit{``ổn định trong thực tế''}: robustness, structural correctness, domain shift và khả năng mở rộng dữ liệu.
    \end{block}
\end{frame}

\begin{frame}{4.3 \; Các hướng nghiên cứu đề xuất (sơ bộ)}
    \begin{itemize}
        \item \textbf{MLLM \textrightarrow{} Data Factory \textrightarrow{} Distillation:} Tận dụng open-source MLLM để tạo dữ liệu multi-task document (OCR + Layout + Parsing), sau đó distill xuống mô hình nhỏ hơn.
        \item \textbf{Domain-specific + Low-resource language:} Tập trung tài liệu chuyên biệt (ví dụ giấy tờ hành chính), ngôn ngữ low-resource (Việt/thiểu số); public quy trình + training data + benchmark.
        \item \textbf{Inject Layout into MLLM:} Khai thác bounding box/class như tín hiệu cấu trúc để tăng độ chính xác cho Document VQA, parsing, và giảm nhầm lẫn ngữ nghĩa.
        \item \textbf{Unified Generative Document Modeling:} Dùng MLLM cho unified tasking (OCR + localization + logical layout + relation), đồng thời tối ưu token/latency bằng fine-tuning và nén mô hình.
        \item \textbf{Vietnamese-focused Backbones:} Xây hoặc pretrain backbone đa phương thức thiên về tài liệu tiếng Việt (độ đa dạng domain cao), tương tự hướng tài nguyên ngôn ngữ kiểu PhoBERT/PhoWhisper.
        \item \textbf{Entity Relation Analysis:} Khoét sâu phân tích quan hệ giữa các thực thể trong tài liệu theo hướng graph/GNN-based để tăng chất lượng hiểu cấu trúc logic.
    \end{itemize}
        \vspace{0.3em}
        	\textit{Mục tiêu chung: vừa có đóng góp học thuật (novelty/publication), vừa đảm bảo khả năng ứng dụng thực tế (production-ready).}
\end{frame}
